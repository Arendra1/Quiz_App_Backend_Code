# Quiz_App_Backend_Code
This is repository for backend code of quiz app

# command for initialising project
npm init -y

# express js
npm i express

# nodemon
npm i -D nodemon

# mongoose
npm i mongoose

# dotenv
npm i dotenv

# jasonwebtoken (JWT)
npm i jsonwebtoken

# bcrypt for password
npm i bcrypt


# nodemailer for sending mails
npm i nodemailer

# cors to handle frontend  CORS errors
npm i cors
